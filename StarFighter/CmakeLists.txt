add_executable(${STAR_FIGHTER_GAME_TARGET_NAME}
	${CMAKE_CURRENT_SOURCE_DIR}/src/main.cpp
)

target_link_libraries(${STAR_FIGHTER_GAME_TARGET_NAME} PUBLIC sfml-audio)
target_link_libraries(${STAR_FIGHTER_GAME_TARGET_NAME} PUBLIC sfml-graphics)
target_link_libraries(${STAR_FIGHTER_GAME_TARGET_NAME} PUBLIC sfml-network)
target_link_libraries(${STAR_FIGHTER_GAME_TARGET_NAME} PUBLIC sfml-system)
target_link_libraries(${STAR_FIGHTER_GAME_TARGET_NAME} PUBLIC sfml-window)

# Function name is CopyLibToTarget. Parameters names of function are LIB_NAME and TARGET_NAME
function(CopyLibToTarget LIB_NAME TARGET_NAME)
	add_custom_command(TARGET ${TARGET_NAME}
	POST_BUILD
	COMMAND
	${CMAKE_COMMAND} -E copy_if_different
	$<TARGET_FILE:${LIB_NAME}>
	$<TARGET_FILE_DIR:${TARGET_NAME}>
	)
endfunction()

CopyLibToTarget(sfml-audio ${STAR_FIGHTER_GAME_TARGET_NAME})
CopyLibToTarget(sfml-graphics ${STAR_FIGHTER_GAME_TARGET_NAME})
CopyLibToTarget(sfml-network ${STAR_FIGHTER_GAME_TARGET_NAME})
CopyLibToTarget(sfml-system ${STAR_FIGHTER_GAME_TARGET_NAME})
CopyLibToTarget(sfml-window ${STAR_FIGHTER_GAME_TARGET_NAME})
